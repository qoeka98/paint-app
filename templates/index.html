from fastapi import FastAPI from fastapi.responses import StreamingResponse import cv2 app = FastAPI() # 웹캠 활성화 cap = cv2.VideoCapture(0) def generate_frames(): """ 웹캠에서 프레임을 읽고 스트리밍 """ while True: success, frame = cap.read() if not success: break _, buffer = cv2.imencode('.jpg', frame) frame_bytes = buffer.tobytes() yield (b'--frame\r\n' b'Content-Type: image/jpeg\r\n\r\n' + frame_bytes + b'\r\n') @app.get("/video_feed") async def video_feed(): """ 실시간 웹캠 스트리밍 엔드포인트 """ return StreamingResponse(generate_frames(), media_type="multipart/x-mixed-replace; boundary=frame") if __name__ == "__main__": import uvicorn uvicorn.run(app, host="0.0.0.0", port=8000, reload=True)
